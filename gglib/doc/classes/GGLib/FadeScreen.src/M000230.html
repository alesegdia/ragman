  def draw
    if @fading
      @alpha+=@speed
      if @alpha > 255
        @alpha=255
      end
      @color.alpha=@alpha
      @image.draw(0, 0, ZOrder::Top+1, 640, 480, @color)
      if @alpha == 255
        endFade
        @widgetID = FadeIn.new(@speed)
      end
    end
  end
