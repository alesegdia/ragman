  def event(type, dat=nil) #do not modify
    if type==Event::MsLeft
      if acceptStickyFocus?
        stickFocus
      end
      onClick
    elsif type==Event::MsRight
      onRightClick
    elsif type==Event::KeyUp
      onKeyUp(dat)
    elsif type==Event::KeyDown
      onKeyDown(dat)
    end
  end
