    def initialize(speed=5)
      @image=Gosu::Image.new($window, $gglroot+"/media/black.bmp", true)
      @speed=speed
      @color=Gosu::Color.new(0x00ffffff)
      @alpha=255
      super(Button::Image, 0, 0, 0, 0)
      wakeUp
      stickFocus
      $window.setFocus(nil)
    end
